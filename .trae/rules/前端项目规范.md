### 2.2 禁止使用的技术

| 禁止技术 | 原因 | 推荐替代 |
|----------|------|----------|
| Vue 2 | 已停止维护 | Vue 3 |
| Options API | 代码组织性差 | Composition API + `<script setup>` |
| Vuex | 已停止维护 | Pinia |
| `any` 类型 | 丢失类型安全 | 明确类型定义 |
| `var` | 函数作用域问题 | `const` / `let` |
| `any` | 丢失类型检查 | Interface / Type |

## 4. 命名规范

### 4.1 文件命名

| 类型 | 规范 | 示例 |
|------|------|------|
| Vue 组件 | 大驼峰或 kebab-case | `UserInfo.vue` / `user-info.vue` |
| TS/JS 文件 | 小驼峰或 kebab-case | `userInfo.ts` / `user-info.ts` |
| 样式文件 | kebab-case | `user-profile.scss` |
| 枚举文件 | 小驼峰 + `Enum` 后缀 | `httpEnum.ts` |
| 工具文件 | 小驼峰或功能名 + `Utils` | `request.ts` / `dateUtils.ts` |
| API 文件 | 小驼峰或按模块目录组织 | `user.ts` / `article/index.ts` |
| Store 文件 | 小驼峰 + `Store` 后缀或按模块 | `user.ts` / `userStore.ts` |

### 4.2 组件命名

```vue
<!-- PascalCase - 推荐 -->
<UserCard />
<ArticleList />

<!-- kebab-case - 也可接受 -->
<user-card />
<article-list />
```

### 4.3 变量命名

| 类型 | 规范 | 示例 |
|------|------|------|
| 普通变量 | 小驼峰 | `userName`, `isLoading` |
| 常量 | 全大写 + 下划线 | `MAX_RETRY_COUNT`, `API_BASE_URL` |
| 布尔变量 | `is` / `has` / `can` 前缀 | `isVisible`, `hasPermission` |
| 数组变量 | 复数名词或加 `List` | `users`, `userList` |
| 对象变量 | 名词或加 `Info` / `Data` | `userInfo`, `articleData` |
| DOM 引用 | 加 `$` 后缀或 `Ref` 后缀 | `inputRef`, `formRef` |

### 4.4 CSS 类名命名

使用 BEM 命名规范或 kebab-case：

```scss
// BEM 规范
.user-card {
  &__header { }
  &__body { }
  &--active { }
}

// kebab-case
.user-card { }
.user-card-header { }
.user-card-body { }
```

## 5. 组件规范

### 5.1 组件结构

```vue
<script setup lang="ts">
// 1. 导入
import { ref, computed, onMounted } from 'vue'
import type { PropType } from 'vue'

// 2. Props 定义
interface Props {
  title: string
  data?: any[]
  onChange?: (val: string) => void
}

const props = withDefaults(defineProps<Props>(), {
  title: '默认标题',
  data: () => []
})

// 3. Emits 定义
const emit = defineEmits<{
  (e: 'update', value: string): void
  (e: 'close'): void
}>()

// 4. 响应式数据
const loading = ref(false)
const localValue = ref('')

// 5. 计算属性
const isValid = computed(() => localValue.value.length > 0)

// 6. 方法
function handleSubmit() {
  emit('update', localValue.value)
}

// 7. 生命周期
onMounted(() => {
  // 初始化
})
</script>

<template>
  <div class="component-name">
    <!-- 模板内容 -->
  </div>
</template>

<style scoped lang="scss">
// 使用 scoped 避免样式污染
.component-name {
  // 样式
}
</style>
```

### 5.2 组件分类

| 类型 | 存放位置 | 说明 |
|------|----------|------|
| 基础组件 | `components/` | 可复用的原子组件 |
| 业务组件 | `pages/` 子目录 | 特定页面使用的组件 |
| 布局组件 | `layouts/` | 页面整体布局 |
| 业务组件 | `views/` 子目录 | 页面级组件 |

### 5.3 Props 定义

```typescript
// ✅ 推荐：使用 TypeScript 接口 + withDefaults
interface Props {
  title: string
  count?: number
  list?: string[]
  onChange?: (val: string) => void
}

const props = withDefaults(defineProps<Props>(), {
  count: 0,
  list: () => []
})

// ✅ 推荐：使用 defineProps + 泛型
const props = defineProps<{
  title: string
  data: UserInfo
}>()

// ❌ 避免：使用数组语法
// const props = defineProps(['title', 'data'])
```

### 5.4 事件命名

```typescript
// ✅ 推荐：kebab-case 事件名
emit('update:value', newValue)
emit('item-click', item)

// ✅ 推荐：使用 v-model 规范
// 组件定义
const props = defineProps<{ modelValue: string }>()
const emit = defineEmits<{ 'update:modelValue': [value: string] }>()

// 组件使用
// <MyInput v-model="value" />
```

### 6.2 Store 定义方式

```typescript
// Admin 项目 - 使用 setup 风格
import { defineStore } from 'pinia'

export const useUserStore = defineStore('user', () => {
  // 状态
  const userInfo = ref<UserInfo>()
  const token = ref<string>()

  // 计算属性
  const isLoggedIn = computed(() => !!token.value)
  const nickname = computed(() => userInfo.value?.nickname)

  // 方法
  async function fetchUserInfo() {
    const { data } = await getUserInfoApi()
    userInfo.value = data
  }

  function logout() {
    token.value = null
    userInfo.value = undefined
  }

  return {
    userInfo,
    token,
    isLoggedIn,
    nickname,
    fetchUserInfo,
    logout
  }
})

// Blog 项目 - 同样使用 setup 风格
import { defineStore } from 'pinia'

const useUserStore = defineStore('user', () => {
  const token = GET_TOKEN()
  const userInfo = shallowRef<UserInfo>()

  const getInfo = async () => {
    const res: any = await getUserInfo()
    if (res.code === 200) {
      userInfo.value = res.data
    }
  }

  return {
    token,
    userInfo,
    getInfo
  }
})

export default useUserStore
```

### 6.3 Store 使用规范

```typescript
// ✅ 推荐：组合式函数方式
const userStore = useUserStore()

// ❌ 避免：在 setup 外使用 store
// const store = useUserStore() // 不要在 setup 顶层直接调用

// 在 setup 或 composable 中使用
function MyComponent() {
  const userStore = useUserStore()
  // ...
}
```

## 7. API 请求规范

### 7.1 Admin 项目 - API 封装

```typescript
// utils/request.ts - Axios 封装
export interface ResponseBody<T = any> {
  code: number
  data?: T
  msg: string
}

export function useGet<R = any, T = any>(
  url: string,
  params?: T,
  config?: RequestConfigExtra
): Promise<ResponseBody<R>> {
  return instancePromise({ url, params, method: RequestEnum.GET, ...config })
}

export function usePost<R = any, T = any>(
  url: string,
  data?: T,
  config?: RequestConfigExtra
): Promise<ResponseBody<R>> {
  return instancePromise({ url, data, method: RequestEnum.POST, ...config })
}

// api/user/index.ts - API 接口定义
export async function userList() {
  return useGet('/user/list').catch(msg => message.warn(msg))
}

export async function userDetail(id: string) {
  return useGet(`/user/details/${id}`)
}

export async function userDelete(ids: string[]) {
  return useDelete('/user/delete', { ids })
}
```

### 7.2 Blog 项目 - API 封装

```typescript
// utils/http.ts - Axios 实例
const http: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_APP_BASE_API ?? '/',
  timeout: 60000
})

// apis/article/index.ts - API 接口定义
export const getArticleDetail = (id: string | string[]) => {
  return http.request({
    url: `/article/detail/${id}`,
    method: 'get'
  })
}

export const addComment = (data: object) => {
  return http.request({
    url: '/comment/auth/add/comment',
    method: 'post',
    data
  })
}
```

### 7.3 API 设计原则

| 原则 | 说明 |
|------|------|
| 按模块组织 | `apis/article/`, `apis/user/` |
| RESTful 风格 | GET 查询、POST 新增、PUT 更新、DELETE 删除 |
| 统一返回格式 | `{ code: number, data: T, msg: string }` |
| 类型安全 | 定义完整的请求和响应类型 |

### 7.4 API 文件组织

```
// ✅ 推荐：按功能模块组织
apis/
```

## 8. TypeScript 规范

### 8.1 类型定义

```typescript
// ✅ 推荐：使用 interface 定义对象类型
interface UserInfo {
  id: number
  username: string
  avatar?: string
  roles: string[]
}

// ✅ 推荐：使用 type 定义联合类型或工具类型
type UserStatus = 'active' | 'inactive' | 'banned'
type PartialUser = Partial<UserInfo>
type RequiredUser = Required<UserInfo>
type PickUser = Pick<UserInfo, 'id' | 'username'>

// ✅ 推荐：定义 API 响应类型
interface ApiResponse<T> {
  code: number
  data: T
  msg: string
}

interface PageResult<T> {
  list: T[]
  total: number
  pageNum: number
  pageSize: number
}

// ✅ 推荐：定义枚举类型
enum RequestMethod {
  GET = 'GET',
  POST = 'POST',
  PUT = 'PUT',
  DELETE = 'DELETE'
}

// ✅ 推荐：定义组件 Props 类型
interface Props {
  title: string
  data: UserInfo[]
  loading?: boolean
}
```

### 8.2 禁止使用

```typescript
// ❌ 禁止：使用 any
function handleData(data: any) {
  // ...
}

// ✅ 推荐：使用 unknown 或具体类型
function handleData(data: unknown) {
  if (isUserInfo(data)) {
    // ...
  }
}

// 或者使用具体类型
function handleData(data: UserInfo) {
  // ...
}

// ❌ 禁止：使用 var
var name = 'test'

// ✅ 推荐：使用 const / let
const name = 'test'
let count = 0
```

### 8.3 类型导出

```typescript
// types/user.ts
export interface UserInfo {
  id: number
  username: string
}

export type UserStatus = 'active' | 'inactive'

// 在组件或 API 中导入
import type { UserInfo, UserStatus } from '@/types/user'
```

## 10. 路由规范

### 10.1 Admin 项目路由

```typescript
// router/static-routes.ts - 静态路由
export const staticRoutes: RouteRecordRaw[] = [
  {
    path: '/login',
    name: 'Login',
    component: () => import('@/pages/common/login.vue')
  },
  {
    path: '/:pathMatch(.*)*',
    name: 'NotFound',
    component: () => import('@/pages/exception/404.vue')
  }
]

// router/dynamic-routes.ts - 动态路由
export const rootRoute: RouteRecordRaw = {
  path: '/',
  component: Layout,
  redirect: '/welcome',
  children: []
}

// 路由守卫
// router/router-guard.ts
router.beforeEach(async (to, from, next) => {
  const userStore = useUserStore()

  if (to.path === '/login') {
    next()
  } else {
    if (userStore.token) {
      next()
    } else {
      next({ path: '/login', query: { redirect: to.fullPath } })
    }
  }
})
```

## 14. 时间规范

### 14.1 核心原则

**前端时间处理应使用 JavaScript 原生的 `Date` 对象或第三方时间库，禁止使用不安全的日期时间处理方式。**

### 14.2 推荐的时间库

| 库名 | 说明 | 适用场景 |
|------|------|----------|
| **Day.js** | 轻量级，API 与 Moment.js 兼容 | ✅ 推荐，简单日期处理 |
| **date-fns** | 模块化，Tree-shaking 友好 | 功能复杂时推荐 |
| **Moment.js** | 功能全面 | ⚠️ 不推荐，体积过大（已停止维护） |

### 14.3 项目中的时间处理

本项目使用 **Day.js** 进行时间处理：

```typescript
import dayjs from 'dayjs'

// Blog 项目使用
import { format } from '@/utils/tool'

// 获取当前时间
const now = dayjs()

// 格式化时间
const formatted = dayjs().format('YYYY-MM-DD HH:mm:ss')

// 解析时间字符串
const parsed = dayjs('2024-01-15 10:30:00', 'YYYY-MM-DD HH:mm:ss')

// 日期计算
const tomorrow = dayjs().add(1, 'day')
const yesterday = dayjs().subtract(1, 'day')
const nextMonth = dayjs().add(1, 'month')
const lastYear = dayjs().subtract(1, 'year')

// 日期比较
dayjs().isBefore(otherDate)
dayjs().isAfter(otherDate)
dayjs().isSame(otherDate)

// 时间差计算
dayjs().diff(otherDate, 'day')    // 相差天数
dayjs().diff(otherDate, 'hour')   // 相差小时
dayjs().diff(otherDate, 'minute') // 相差分钟
```

### 14.4 禁止使用的时间处理方式

| 禁止方式 | 原因 | 推荐替代 |
|----------|------|----------|
| `new Date()` 直接操作 | API 复杂，易出错 | Day.js |
| `moment.js` | 体积过大（已停止维护） | Day.js / date-fns |
| 日期字符串手动拼接 | 容易出现格式错误 | 使用库函数 |
| `setTimeout` / `setInterval` 处理时间 | 不精确 | 使用库函数 |

### 14.5 日期格式字符串规范

| 格式 | 说明 | 示例 |
|------|------|------|
| `YYYY` | 4位年份 | 2024 |
| `MM` | 2位月份 | 01-12 |
| `DD` | 2位日期 | 01-31 |
| `HH` | 24小时制小时 | 00-23 |
| `hh` | 12小时制小时 | 01-12 |
| `mm` | 分钟 | 00-59 |
| `ss` | 秒 | 00-59 |
| `A` | 上午/下午 | AM / PM |
| `dddd` | 星期几 | Sunday |

### 14.6 常用格式化示例

```typescript
// 日期时间格式化
dayjs().format('YYYY-MM-DD HH:mm:ss')        // 2024-01-15 10:30:00
dayjs().format('YYYY/MM/DD HH:mm:ss')        // 2024/01/15 10:30:00
dayjs().format('YYYY年MM月DD日 HH:mm')        // 2024年01月15日 10:30

// 日期格式化
dayjs().format('YYYY-MM-DD')                  // 2024-01-15
dayjs().format('MM/DD/YYYY')                  // 01/15/2024

// 时间格式化
dayjs().format('HH:mm:ss')                    // 10:30:00
dayjs().format('hh:mm:ss A')                 // 10:30:00 AM

// 相对时间
dayjs().from(dayjs().subtract(1, 'day'))     // a day ago
dayjs().fromNow()                            // a few seconds ago
dayjs().to(dayjs().add(1, 'day'))            // in a day
```

### 14.7 后端接口时间处理

#### 14.7.1 时间格式约定

前端与后端交互时，时间格式统一使用 **ISO 8601** 格式或 **Unix 时间戳**：

```typescript
// ISO 8601 格式（推荐）
'2024-01-15T10:30:00.000Z'
'2024-01-15T10:30:00+08:00'

// Unix 时间戳（毫秒）
1705289400000

// 发送请求时格式化
const timeValue = dayjs().format('YYYY-MM-DDTHH:mm:ss.SSS[Z]')

// 解析后端返回的时间
const date = dayjs(response.data.createTime)
```

#### 14.7.2 时间显示规范

|------|------|------|

| 短日期 | `YYYY-MM-DD` | 2024-01-15 |

### 14.8 前端时间处理工具封装

#### 14.8.1 Blog 项目工具封装

```typescript
// utils/tool.ts
import dayjs from 'dayjs'
import relativeTime from 'dayjs/plugin/relativeTime'
import 'dayjs/locale/zh-cn'

dayjs.extend(relativeTime)
dayjs.locale('zh-cn')

/**
 * 格式化时间
 */
export function formatTime(date: string | Date | number, formatStr = 'YYYY-MM-DD HH:mm:ss'): string {
  return dayjs(date).format(formatStr)
}

/**
 * 相对时间
 */
export function formatRelativeTime(date: string | Date | number): string {
  return dayjs(date).fromNow()
}

/**
 * 获取相对时间描述
 */
export function getTimeDesc(date: string | Date | number): string {
  const now = dayjs()
  const target = dayjs(date)
  const diffMinutes = now.diff(target, 'minute')
  const diffHours = now.diff(target, 'hour')
  const diffDays = now.diff(target, 'day')

  if (diffMinutes < 1) return '刚刚'
  if (diffMinutes < 60) return `${diffMinutes}分钟前`
  if (diffHours < 24) return `${diffHours}小时前`
  if (diffDays < 7) return `${diffDays}天前`
  return dayjs(date).format('YYYY-MM-DD')
}
```

#### 14.8.2 使用示例

```typescript
import { formatTime, getTimeDesc } from '@/utils/tool'

// 格式化时间
const formatted = formatTime('2024-01-15 10:30:00')
// 结果: 2024-01-15 10:30:00

// 相对时间
const relative = getTimeDesc(Date.now() - 5 * 60 * 1000)
// 结果: 5分钟前

// 在组件中使用
<span>{{ formatTime(article.createTime, 'YYYY-MM-DD') }}</span>
<span>{{ getTimeDesc(comment.createTime) }}</span>
```
