**3.2 auth.ts - 登录接口适配：**

文件：`apps/web-antd-bakend/src/api/core/auth.ts`

```typescript
/**
 * 登录 - 使用 FormData 和 baseRequestClient
 * 原因：
 * 1. 后端 Spring Security 表单登录要求 FormData 格式
 * 2. 登录接口不返回 ResultData 格式（没有 data 字段）
 * 3. token 字段名是 "token" 而非 "accessToken"
 */
export async function loginApi(data: AuthApi.LoginParams) {
  const formData = new FormData();
  if (data.username) formData.append('username', data.username);
  if (data.password) formData.append('password', data.password);

  // 使用 baseRequestClient 避免自动 unwrap data 字段
  const result = await baseRequestClient.post<{
    code: string;
    token: string;
    msg: string;
    expire: string;
  }>('/auth/login', formData, {
    headers: { 'Content-Type': 'multipart/form-data' },
  });

  // 字段转换: token -> accessToken
  return {
    accessToken: result.token,
  };
}
```

**3.3 user.ts - 用户信息接口适配：**

文件：`apps/web-antd-bakend/src/api/core/user.ts`

```typescript
/**
 * 获取用户信息
 * 后端路径: /user/auth/info
 * 后端返回: ResultData<UserAccountVO>
 * 字段映射:
 *   nickname -> realName
 *   id -> userId (String)
 *   intro -> desc
 */
export async function getUserInfoApi() {
  // requestClient 会自动 unwrap data 字段
  const data = await requestClient.get<{
    id?: number;
    nickname: string;
    username: string;
    avatar: string;
    intro?: string;
    roles: string[];
  }>('/user/auth/info');

  return {
    avatar: data.avatar || '',
    realName: data.nickname || data.username,
    roles: data.roles || [],
    userId: String(data.id || 0),
    username: data.username,
    desc: data.intro || '',
    homePath: '/dashboard',
    token: '',
  } as UserInfo;
}
```

**4. 改造后流程：**

```
前端登录流程：
1. 调用 loginApi(username, password)
   → POST /auth/login (FormData)
   → baseRequestClient (不做 unwrap)
   → 返回: { accessToken: "..." }

2. 路由守卫拦截 → 调用 getUserInfoApi()
   → GET /user/auth/info (携带 Bearer Token)
   → requestClient (自动 unwrap data)
   → 返回: UserInfo (字段已映射)

3. 调用 getAccessCodesApi()
   → GET /auth/codes
   → 返回权限码列表
```

- 使用 `baseRequestClient` 处理非 ResultData 格式的响应
- 使用 `requestClient` 处理 ResultData 格式的响应（自动 unwrap data）
